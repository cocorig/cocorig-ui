# ColorPalette

## 사용 방법

<Source language="js" dark code={`import { COLOR_PALLETTE } from 'cocorig-ui';`} />

export function colorItemList(items) {
  return Object.entries(items).map(([key, color]) => {
    return <ColorItem title={key} subtitle="" colors={color} />;
  });
}

## 색상

<ColorPalette>{colorItemList(COLOR_PALLETTE)}</ColorPalette>

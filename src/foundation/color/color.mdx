# Colors

export function colorItemList(items) {
  return Object.entries(items).map(([key, color]) => {
    if (typeof color === 'object') {
      return <ColorItem title={key} subtitle="" colors={color} />;
    }
    if (typeof color === 'string') {
      return <ColorItem title={key} subtitle="" colors={{ [key]: color }} />;
    }
  });
}
